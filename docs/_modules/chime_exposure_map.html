[docs]def query_location_az_alt(date, ra, dec):
    """
    Calculate az, alt given a date, ra, and dec

    Parameters
    ----------

    date : string
       A string containing the date and time in
       "2018/01/25 15:00:30" order, where the
       time is given in UTC.

    ra : ephem.Degrees
        The right ascension of an object on the sky.
        For example, ephem.Degrees('00:00:00.000')

    dec : ephem.Degrees
        The right ascension of an object on the sky.

    Returns
    -------

    tuple
        A float 2-tuple: (az, alt), in degrees,
        representing the azimuth and altitude
        of an object on the sky

    """

    chime = generate_chime_ephem()
    chime.date = date

    test_body = ephem.FixedBody()
    test_body._ra = ra
    test_body._dec = dec

    test_body.compute(chime)

    az = 180.*float(test_body.az)/(np.pi)
    alt = 180.*float(test_body.alt)/(np.pi)

    return az, alt



[docs]def exposure_lightcurve(
        ra_degree,
        dec_degree,
        start_date,
        end_date,
        resolution,
        beam_ids,
        frequencies,
        beam_positions=None):
    """
    Generate a timestream of exposure to a specific point

    This uses the max sensitivity over the beams, rather than
    a sum of sensitivities.

    Parameters
    ----------

    ra_degree : float
        The right ascension of an object on the sky.
        In degrees.

    dec_degree : float
        The declination of an object on the sky.
        In degrees.

    start_date : datetime.datetime
        The start of the timestream.

    end_date : datetime.datetime
        The end of the timestream.

    resolution : datetime.timedelta
        The sampling resolution of the timestream. This,
        and the length of the observation, determines
        how many samples will be returned in total

    beam_ids : array_like
        Shape (N,). List of beams in terms of their IDs which
        will be included in the timestream. Length
        N (integers can be in [0,1023]).

    frequencies : array_like
        Shape (F,). List of which frequencies to calculate sensitivity
        for. Frequencies are floats given in terms of MHz.

    Returns
    -------

    numpy.ndarray
        A float numpy.ndarray: (T, F), where T is the number of samples,
        and F is the number of frequencies. The floats represent
        the max sensitivity over all beams to the point
        in the sky. The first sample
        is at the start date, and the last sample is
        at or before the end_date. Samples are spaced
        according to the resolution provided.

    """
    #TODO: Update this docstring
    if beam_positions is None:
        beam_positions = get_beam_centers()



    curr_date = start_date

    beam_model = FFTFormedSincBeamModel(BEAM_MODEL_CONFIG)

    samples = int((end_date - start_date).total_seconds() / resolution.total_seconds() + 1)
    time_of_samples = [curr_date + i*resolution for i in range(samples)]
    sensitivity = np.zeros((samples, len(frequencies)))

    for i in range(int(samples)):

        curr_date = time_of_samples[i]

        position = beam_model.get_position_from_equatorial(
            ra_degree,
            dec_degree,
            curr_date)

        all_sensitivites = beam_model.get_sensitivity(
            beam_ids,
            positions=np.array([position]),
            freqs=frequencies,
            beam_positions=beam_positions)
        max_sensitivity = np.max(all_sensitivites, axis=0)

        sensitivity[i] = max_sensitivity

    return np.array(sensitivity)

[docs]def generate_exposure_map(
        start_date,
        end_date,
        spatial_resolution,
        beam_ids,
        frequencies,
        detection_threshold,
        ra_lower,
        ra_upper,
        dec_lower,
        dec_upper):
    """
    Generate a timestream of exposure to a specific point

    This uses the max sensitivity over the beams, rather than
    a sum of sensitivities.

    Parameters
    ----------

    start_date : datetime.datetime
        The start of the map generation

    end_date : datetime.datetime
        The end time of the map generation.

    spatial_resolution : float
        The sampling resolution of the map in degrees.

    beam_ids : array_like
        Shape (N,). List of beams in terms of their IDs which
        will be included in the timestream. Length
        N (integers can be in [0,1023]).

    frequencies : array_like
        Shape (F,). List of which frequencies to calculate sensitivity
        for. Frequencies are floats given in terms of MHz.

    detection_threshold : float
        The sensitivity (0-1) that is considered for a "detection",
        meaning the time spent with sensitivity > _ for any beam
        is considered "detecting time," and is integrated over.

    ra_lower : float
        The right ascension lower-limit of the map
        In degrees.

    ra_upper : float
        The right ascension upper-limit of the map
        In degrees.

    dec_lower : float
        The declination lower-limit of the map
        In degrees.

    dec_upper : float
        The declination upper-limit of the map
        In degrees.

    beam_ids : array_like
        Shape (N,). List of beams in terms of their IDs which
        will be included in the timestream. Length
        N (integers can be in [0,1023]).

    frequencies : array_like
        Shape (F,). List of which frequencies to calculate sensitivity
        for. Frequencies are floats given in terms of MHz.

    Returns
    -------

    numpy.ndarray
        A float numpy.ndarray: (R, D, F), where R is the number of
        right ascension samples on the map, D is the number of
        declination samples on the map, and F is the number of
        frequencies. The floats in the array represent
        the minutes over all beams to the point
        in the sky. The first sample
        is at the start date, and the last sample is
        at or before the end_date. Samples are spaced
        according to the resolution provided.

    """

    exposure_map = np.zeros((
        len(np.arange(ra_lower, ra_upper, spatial_resolution)),
        len(np.arange(dec_lower, dec_upper, spatial_resolution)),
        len(frequencies)))

    time_resolution = datetime.timedelta(minutes=1)
    for i, ra in enumerate(np.arange(ra_lower, ra_upper, spatial_resolution)):
        for j, dec in enumerate(np.arange(dec_lower, dec_upper, spatial_resolution)):
            lightcurve = exposure_lightcurve(
                ra,
                dec,
                start_date,
                end_date,
                datetime.timedelta(minutes=1),
                beam_ids,
                frequencies)

            good_samples = np.sum(lightcurve > detection_threshold)
            exposure_map[i, j, 0] = good_samples * 1.
            break
        break
    poses = []
    for i, ra in enumerate(np.arange(ra_lower, ra_upper, spatial_resolution)):
        for j, dec in enumerate(np.arange(dec_lower, dec_upper, spatial_resolution)):
            if i == 0 and j == 0:
                continue
            poses.append((ra, dec))


    len_ra = len(np.arange(ra_lower, ra_upper, spatial_resolution))
    len_dec = len(np.arange(dec_lower, dec_upper, spatial_resolution))
    new_exp_maps = np.zeros(len(poses))
    #def update_exp_map(result):
        #new_exp_maps[result[1]] = result[0]

    #myclass = call_for_exp_map(start_date, end_date, time_resolution, beam_ids, frequencies)
    def f3(inputted):
        lightcurve = exposure_lightcurve(
            inputted[0],
            inputted[1],
            start_date,
            end_date,
            resolution,
            beam_ids,
            frequencies)
        good_samples = np.sum(lightcurve > detection_threshold)
        return good_samples
    def call_back(result):
        new_exp_maps[result[1]] = result[0]

    p = Pool(8)
    for i in range(len(poses)):
        #def f4(start_date, end_date, resolution, beam_ids, frequencies, detection_threshold, inputted, index):
        p.apply_async(f4, args=(start_date, end_date, time_resolution, beam_ids, frequencies, detection_threshold, poses[i], i), callback=call_back)
    p.close()
    p.join()

    for i, ra in enumerate(np.arange(ra_lower, ra_upper, spatial_resolution)):
        for j, dec in enumerate(np.arange(dec_lower, dec_upper, spatial_resolution)):
            if i == 0 and j == 0:
                continue
            exposure_map[i, j, 0] = new_exp_maps[i*len_dec + j - 1] * 1.

    return exposure_map
