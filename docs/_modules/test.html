        """ Assert that the observer object has valid sunrise time """
        chime_telescope_observer = generate_chime_ephem()
        chime_telescope_observer.date = '2018/1/18'

        # Check if sunrise is correct
        sun_location = ephem.Sun()
        sun_location.compute(chime_telescope_observer)
        sunrise_time_utc = chime_telescope_observer.previous_rising(

            ephem.Date(sunrise_time_utc - 8*ephem.hour))

        self.assertTrue(
            str(sunrise_time_pst)[10:15] == '07:46'
            )

[docs]    def test_sensitivity_calculation(self):
        """ Test sensitivity is max in center of one beam """

        beam_model = FFTFormedSincBeamModel(BEAM_MODEL_CONFIG)

        center_beam = [0., 90.]
        off_beam = [
            0+BEAM_MODEL_CONFIG['beam_sep_x'], #EW
            90.+BEAM_MODEL_CONFIG['beam_sep_x']] #NS

        beams = list(range(256))
        sensitivies = beam_model.get_sensitivity(
            beams,
            positions=np.array([center_beam, off_beam]),
            freqs=[600])
        # Center beam > off-beam
        self.assertGreater(np.max(sensitivies[0]), np.max(sensitivies[1]))

[docs]    def test_timestream_validitity(self):
        """
        Test various expected properties of the timestream

        These include:
            - Most of the time, 0 sensitivity is expected for low latitude
            - Some of the time, high sensitivity is expected for low lat.
        """

        self.generate_default_lightcurve()
        number_samples = self.lightcurve.shape[0]
        lightcurve_peak = np.max(self.lightcurve)
        zero_sample_indices = np.argwhere(
            self.lightcurve < lightcurve_peak*0.01)
        number_zero_samples = (self.lightcurve[zero_sample_indices]).shape[0]
        number_nonzero_samples = number_samples - number_zero_samples

        self.assertGreater(number_samples, number_zero_samples)

        # Should have more almost-zero samples than nonzero
        self.assertGreater(number_zero_samples, number_nonzero_samples)

[docs]    def test_double_time_doubles_exposure(self):
        """ Test that doubling time doubles exposure """
        self.spatial_resolution = 9.
        self.dec_injection = 40. + 40./60.
        self.ra_lower = (15. + 44.75/60.)/24.*360. - 0.5
        self.ra_upper = self.ra_lower + 1.
        self.end_date = self.start_date + datetime.timedelta(seconds=1000)
        self.generate_default_map()
        short_time_exposure = np.sum(self.exposure_map)
        self.end_date = self.start_date + datetime.timedelta(seconds=2000)
        self.generate_default_map()
        long_time_exposure = np.sum(self.exposure_map)

        # Verify the short time is ~2 smaller
        self.assertGreater(long_time_exposure, 0.00)
        np.testing.assert_almost_equal(
            short_time_exposure*2./long_time_exposure,
            1,
            0)

[docs]    def test_snr_output_generic(self):
        """ Expect a generic source similar to B0329 to have a large SNR """
        frequencies = np.linspace(400, 800, 5) #MHz
        constant_flux = 15000*np.ones_like(frequencies) #mJy
        width = 3 #ms
        dm = 60
        snr_lightcurve = expected_snr_curve_generic(
            frequencies=frequencies,
            fluxes=constant_flux,
            width=width,
            dm=dm,
            ra=14.,
            dec=55.
            )
        max_snr = np.max(snr_lightcurve)
        self.assertGreater(max_snr, 100)
        zero_time = len(snr_lightcurve[snr_lightcurve < 10])
        nonzero_time = len(snr_lightcurve) - zero_time
        self.assertGreater(zero_time, nonzero_time)
