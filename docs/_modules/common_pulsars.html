import pandas as pd


[docs]def get_pulsar(name):
    """
    Return a dict for a pulsar by name (string),
    using the data in clean_atnf_cat.pkl, which
    is generated from the ATNF catalog.

    Parameters
    ----------

    name : string
        The name of the pulsar (case-sensitive), such
        as "B0329+54"

    Returns
    -------

    dict
        The pulsar parameters as listed in clean_atnf_cat.pkl.

    """
    atnf_cat = pd.read_pickle('clean_atnf_cat.pkl')
    pulsar = atnf_cat[atnf_cat['name'] == name]
    if len(pulsar) == 0:
        raise KeyError("Pulsar not found in cleaned catalog")

    return pulsar.to_dict(orient='records')[0]
