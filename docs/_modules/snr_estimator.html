[docs]def estimate_n_events(
        snr_lightcurve, resolution_minutes, period, snr_threshold=10):
    """
    Calculate the number of expected events seen by CHIME
    from a pulsar given its expected SNR over time.

    This uses a simple formula in that every sample in snr_lightcurve
    that is above the snr_threshold has all of the pulses
    of the pulsar detected by CHIME.

    Parameters
    ----------

    snr_lightcurve : array_like
        The expected SNR of a pulsar's pulses
        over time in an array of shape (T,)
        where T is the number of time samples.

    resolution_minutes : float
        The number of minutes between each datapoint
        in snr_lightcurve. 

    period : astropy.units.Quantity (float)
        The period of the pulsar in units of time.

    snr_threshold : float
        The minimum SNR in the lightcurve required
        for the pulsar to be visible.

    Returns
    -------

    int
        The number of events expected to be observed for
        the pulsar over the snr_lightcurve.

    """

    # Number of events above 10 SNR
    bursts_per_data_point = ((resolution_minutes * u.min)/period).si
    n_events = len(snr_lightcurve[snr_lightcurve > 10.])*bursts_per_data_point
    return n_events



[docs]def get_system_temperature(frequencies):
    """
    Estimate the effective system temperature at every frequency
    passed.

    Parameters
    ----------

    frequency : astropy.units.Quantity (array_like)
        Frequencies to calculate the system temperature at,
        of shape (F,).

    Returns
    -------

    astropy.units.Quantity (array_like)
        The system temperature at each frequency, of
        shape (F,).

    """
    Tfeedloss = 7.*u.K
    Tant = 20.*u.K + 7.*u.K*(frequencies-u.MHz*400)/(u.MHz*400)#Estimated from fig 9 in email with Seth
    Tspillover = 1.*u.K + 3.*u.K*(frequencies-u.MHz*400)/(u.MHz*400) #Estimated from fig 3 of Bandura
    Tsys = Tant+Tfeedloss+Tspillover
    return Tsys




[docs]def linear_interpolate(x, x0, x1, y0, y1):
    """
    Estimate a function using linear interpolation
    using two reference points: (x0, y0), and (x1, y1).
    

    Parameters
    ----------
    
    x : array_like
        Values of the function to estimate
        from the linear interpolation. Of shape
        (X,).

    x0 : float
        Starting reference point.

    x1 : float
        End reference point.

    y0 : float
        Value of the function being
        estimated at the starting reference point.

    y1 : float
        Value of the function being
        estimated at the end reference point.

    Returns
    -------

    array_like
        Estimated values of the function of
        shape (X,) at each value of x.
        
    """
    return y0 + (y1-y0)*(x-x0)/(x1-x0)



[docs]def expected_snr_curve_pulsar(
        pulsar,
        number_frequency_samples=5,
        beams=None,
        beam_centers=None,
        start_date=datetime.datetime(year=2018, month=2, day=10, hour=0),
        end_date=datetime.datetime(year=2018, month=2, day=11, hour=0),
        resolution_minutes=2):
    """
    Calculate the expected SNR seen by CHIME of some pulsar
    given parameters over some time range.

    Parameters
    ----------

    pulsar : dict
        A dictionary containing values
        for a specific pulsar from the ATNF
            catalog, like 'w50', 's400', 's1400',
            'p0', 'ra', 'dec', and 'dm'.

    number_frequency_samples : int
        The number of frequency samples between
        400 and 800 MHz to sample from. Higher
        means more accurate but more computation.
        
    beams : array_like 
        Integer list indicating which beams are used. The
        maximum SNR is chosen at each time sample over all beams.

    beam_centers : array_like
        Of shape (256,) indicating the centers of each beam. The
        default is to use the ones defined in
        load_beam_centers.get_beam_centers. 

    start_date : datetime.datetime
        The starting time of the SNR lightcurve. Default
        is February 10th, 2018 00:00 UTC.

    end_date : datetime.datetime
        The end time of the SNR lightcurve. Default
        is February 11th, 2018 00:00 UTC.

    resolution_minutes : float
        The number of minutes for each time sample.
        The default is 2 minutes.

    Returns
    -------

    array_like
        Array of shape (T, F), where T is the number of time
        samples, and F is the number of frequencies past. Each
        datapoint represents the expectation SNR of pulses at
        that point in time.

    """

    if beams is None:
        beams = list(range(1024))

    resolution = datetime.timedelta(minutes=resolution_minutes)

    if beam_centers is None:
        beam_centers = get_beam_centers()

    start_frequency = 400
    end_frequency = 800
    frequencies = np.linspace(
            start_frequency,
            end_frequency,
            number_frequency_samples)*u.MHz

    dec_injection = pulsar['dec']
    ra_injection = pulsar['ra']/24.*360.
    period = pulsar['p0'] *u.s
    fifty_percentile_width = pulsar['w50']*u.ms
    duty_cycle = (fifty_percentile_width/period).si

    #1.*u.mJy/duty_cycle
    fluxes = linear_interpolate(
        x=frequencies,
        x0=400*u.MHz,
        x1=1400*u.MHz,
        y0=float(pulsar['s400']),
        y1=float(pulsar['s1400']))/duty_cycle

    return expected_snr_curve_generic(
            frequencies=frequencies.to('MHz').value,
            fluxes=fluxes.value,
            width=pulsar['w50'],
            dm=pulsar['dm'],
            ra=ra_injection,
            dec=dec_injection,
            beams=beams,
            beam_centers=beam_centers,
            start_date=start_date,
            end_date=end_date,
            resolution_minutes=resolution_minutes)




[docs]def expected_snr_curve_generic(
            frequencies,
            fluxes,
            width,
            dm,
            ra,
            dec,
            beams=None,
            beam_centers=None,
            start_date=datetime.datetime(year=2018, month=2, day=10, hour=0),
            end_date=datetime.datetime(year=2018, month=2, day=11, hour=0),
            resolution_minutes=2):
    """
    Calculate the expected SNR seen by CHIME of some event of
    given parameters over some time range.

    Parameters
    ----------

    frequencies : array_like
        List of frequencies (floats in terms of MHz) for
        which to use as part of the SNR calculation. An example
        would be [400, 600, 800] which is coarse sampling of
        CHIME's bandwidth. Of shape (F,)

    fluxes : array_like
        Fluxes (floats in terms of mJy) of the event
        at each frequency given. This is of shape (F,).

    width : float
        The width of the burst in ms. NOT an astropy.units.Quantity.

    dm : float
        The DM of the burst in pc/cm^3.

    ra : float
        The right ascension of the event in the sky in decimal hours.

    dec : float
        The declination of the event in the sky in decimal degrees.

    beams : array_like 
        Integer list indicating which beams are used. The
        maximum SNR is chosen at each time sample over all beams.

    beam_centers : array_like
        Of shape (256,) indicating the centers of each beam. The
        default is to use the ones defined in
        load_beam_centers.get_beam_centers. 

    start_date : datetime.datetime
        The starting time of the SNR lightcurve. Default
        is February 10th, 2018 00:00 UTC.

    end_date : datetime.datetime
        The end time of the SNR lightcurve. Default
        is February 11th, 2018 00:00 UTC.

    resolution_minutes : float
        The number of minutes for each time sample.
        The default is 2 minutes.

    Returns
    -------

    array_like
        Array of shape (T, F), where T is the number of time
        samples, and F is the number of frequencies past. Each
        datapoint represents the expectation SNR at
        that point in time.

    """

    if beams is None:
        beams = list(range(1024))

    resolution = datetime.timedelta(minutes=resolution_minutes)

    if beam_centers is None:
        beam_centers = get_beam_centers()

    frequencies = frequencies*u.MHz
    Tsys = get_system_temperature(frequencies)

    #TODO: Clean up the injection thing
    dec_injection = dec
    ra_injection = ra/24.*360.
    width = width*u.ms

    #1.*u.mJy/duty_cycle
    fluxes = fluxes*u.mJy

    #TODO: Use great circle angle in the projection
    projection = np.cos((49.5-dec_injection)/180.*np.pi)
    snr_multiplier = (CHIME_GAIN*fluxes*(2.*width*CHIME_BANDWIDTH)**(0.5)).si*projection

    lightcurve = exposure_lightcurve(
        ra_injection,
        dec_injection,
        start_date,
        end_date,
        resolution,
        beams,
        frequencies,
        beam_centers)

    deg_to_rad = lambda deg: deg/180.*np.pi
    Tsky = u.K*get_temp(deg_to_rad(ra_injection), deg_to_rad(dec_injection), (frequencies/(1*u.MHz)).si)
    snr_lightcurve = (lightcurve*snr_multiplier[None, :]/(Tsky+Tsys)).si


    # Calculate SNR expected over time
    total_lightcurve = np.sqrt(np.average(np.square(snr_lightcurve), axis=1))

    # Return lightcurve SNR
    return total_lightcurve.value
